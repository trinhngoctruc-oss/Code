```
    │   │   ├── index.js
    │   │   └── styles.css
    │   └── package.json
    └── backend/
        ├── app/
        │   ├── __init__.py
        │   ├── routes.py
        │   ├── models.py
        │   └── templates/
        ├── tests/
        │   └── test_app.py
        ├── requirements.txt
        └── run.py
```
